CREATE TABLE orders (
    order_id VARCHAR(50) NOT NULL,
    s3_raw VARCHAR(255),
    s3_pdf VARCHAR(255),
    source VARCHAR(50),
    payload SUPER,        -- nested JSON storage
    created_at TIMESTAMP DEFAULT GETDATE(),
    PRIMARY KEY(order_id)
);
